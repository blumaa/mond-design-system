## ğŸŒ™ Theme Support

Components automatically support light and dark themes:

```tsx
import { useState } from 'react';
import { ThemeProvider } from '@mond-design-system/theme';

import { Text, Heading, Button } from '@mond-design-system/theme';

function App() {
  const [colorScheme, setColorScheme] = useState<'light' | 'dark'>('light');

  return (
    <ThemeProvider colorScheme={colorScheme}>
      <div>
        <Heading level={1}>Welcome</Heading>
        <Text semantic="secondary">
          Content automatically adapts to the current theme
        </Text>
        <Button onClick={() => setColorScheme(prev => prev === 'light' ? 'dark' : 'light')}>
          Toggle Theme
        </Button>
      </div>
    </ThemeProvider>
  );
}
```
